            iterNode.next = node;
        }
        return reverse(dummyHead.next);
    }

    public ListNode reverse(ListNode head) {
        if (head == null || head.next != null) {
            return head;
        }
        ListNode reverseHead = reverse(head.next);
        head.next.next = head;
        head.next = null;
        return reverseHead;
    }
